SELECT I.SUMMARY, I.DESCRIPTION, IL.NAME
FROM ISSUES I
JOIN issueslists IL ON I.ISSUESLIST_ID;

// cz.2

SELECT I.SUMMARY, I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
JOIN USERS U ON U.ID;

// cz.3
SELECT U.ID, U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_NUMBER 
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.ID
HAVING COUNT(*) > 0
